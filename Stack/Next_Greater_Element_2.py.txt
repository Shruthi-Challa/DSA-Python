class Solution(object):
    def nextGreaterElements(self, nums):
        stack = []
        n=len(nums)
        res = [-1]*n

        for i in range(2 * n):
            idx = i % n

            while stack and nums[idx] > nums[stack[-1]]:

                res[stack.pop()]= nums[idx]

            if i < n:
                stack.append(idx)

        return res
