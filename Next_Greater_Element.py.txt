class Solution(object):
    def nextGreaterElement(self, nums1, nums2):

        stack = []
        hashmap = {}

        for num in nums2:

            while stack and num >= stack[-1]:

                hashmap[stack.pop()]= num

            stack.append(num)

        while stack:

            hashmap[stack.pop()] = -1

        res = []

        for num in nums1:

            res.append(hashmap[num])

        return res
